对二分查找的理解

// [left, right]
// [left, len(nums)] 是符合更新right的那个if条件的集合
// [0, right] 是符合更新left的那个if条件的集合
func Bs1(nums []int, target int) int {
	left, right := 0, len(nums)-1 // [left, right]
	for left <= right {
		mid := left + (right-left)/2
		if nums[mid] < target { // or <=,>,>=
			left = mid + 1
		} else {
			right = mid - 1
		}
	}
	return left // or right
	// lower_bound 时,return left
}

// [left, right)
// [left, len(nums)] 是符合更新right的那个if条件的集合
// [0, right) 是符合更新left的那个if条件的集合
func Bs2(nums []int, target int) int {
	left, right := 0, len(nums) // [left, right)
	for left < right {
		mid := left + (right-left)/2
		if nums[mid] < target { // or <=,>,>=
			left = mid + 1
		} else {
			right = mid
		}
	}
	return left // or right
	// lower_bound 时,return left or right
}

// (left, rihgt)
// (left, len(nums)] 是符合更新right的那个if条件的集合
// [0, right) 是符合更新left的那个if条件的集合
func Bs3(nums []int, target int) int {
	left, right := -1, len(nums) //(left, rihgt)
	for left+1 < right {
		mid := left + (right-left)/2
		if nums[mid] < target { // or <=,>,>=
			left = mid
		} else {
			right = mid
		}
	}
	return left // or right
	// lower_bound 时,return right
}
