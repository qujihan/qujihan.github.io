那些我重装Windows/MacOS/Ubuntu时会做的事情

Windows

跳过联网激活使用本地账号

在连接网络的界面按住Shift+F10调出CMD输入OOBE\BYPASSNRO

安装软件

# Needed to run a remote script the first time
Set-ExecutionPolicy RemoteSigned -Scope CurrentUser 

# set scoop dir is C:\Scoop
md C:\Scoop 
irm get.scoop.sh -outfile 'install.ps1'
.\install.ps1 -ScoopDir 'C:\Scoop' -ScoopGlobalDir 'C:\Scoop' -NoProxy

scoop install sudo git neovim 

scoop bucket add extras
scoop install obsidian wechat potplayer trafficmonitor snipaste screentogif powertoys wezterm

scoop bucket add nerd-fontasdfs
sudo scoop install -g FiraCode-NF-Mono JetBrainsMono-NF-Mono

scoop bucket add java
scoop install gcc make cmake rust python go openjdk

scoop bucket add im-select https://github.com/daipeihust/im-select
scoop install im-select

# winget install
winget install  Microsoft.PowerShell Microsoft.VisualStudioCode

# for golang
go install golang.org/x/tools/gopls@latest

MacOS

安装软件

# install brew
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"


brew install git neovim lazygit sevenzip stow

brew install --cask wechat qq alacritty weztrem emacs visual-studio-code docker tencent-lemon bob snipaste squirrel microsoft-edge onedrive obsidian homebrew/cask-fonts/font-jetbrains-mono-nerd-font

# im-select
brew tap daipeihust/tap && brew install im-select

# for shell
brew install romkatv/powerlevel10k/powerlevel10k
brew install zsh-autosuggestions zsh-syntax-highlighting zsh-completions zsh-git-prompt

# for golang
go install golang.org/x/tools/gopls@latest

Ubuntu

设置ssh


# set ssh
ssh-keygen -t ed25519 -C "qujihan@163.com"
cat .ssh/id_ed25519.pub
ssh -T git@github.com

# install NVIM v0.8.3
curl -LO https://github.com/neovim/neovim/releases/download/stable/nvim-linux64.deb
apt install ./nvim-linux64.deb
