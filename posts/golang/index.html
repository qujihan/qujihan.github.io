自定义ListNode

type ListNode struct {
	Val  int
	Next *ListNode
}
func main(){
    list := []int{3, 2, 0, -4}
    head := &ListNode{Val: list[0]}
    tail := head
    for i := 1; i < len(list); i++ {
        tail.Next = &ListNode{Val: list[i]}
        tail = tail.Next
    }
}
